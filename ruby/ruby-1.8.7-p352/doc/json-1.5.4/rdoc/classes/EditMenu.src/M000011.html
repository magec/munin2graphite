      def create
        title = MenuItem.new('Edit')
        title.submenu = menu
        add_item('Copy', ?c, &method(:copy))
        add_item('Paste', ?v, &method(:paste))
        add_separator
        add_item('Find', ?f, &method(:find))
        add_item('Find Again', ?g, &method(:find_again))
        add_separator
        add_item('Sort', ?S, &method(:sort))
        title
      end
