  def Rufus.to_duration_string(seconds, options={})

    return (options[:drop_seconds] ? '0m' : '0s') if seconds <= 0

    h = to_duration_hash seconds, options

    s = DU_KEYS.inject('') { |r, key|
      count = h[key]
      count = nil if count == 0
      r << "#{count}#{key}" if count
      r
    }

    ms = h[:ms]
    s << ms.to_s if ms

    s
  end
