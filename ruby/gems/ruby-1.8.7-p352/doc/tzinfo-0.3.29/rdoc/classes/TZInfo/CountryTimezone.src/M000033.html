    def ==(ct)
      ct.respond_to?(:identifier) && ct.respond_to?(:latitude) &&
      ct.respond_to?(:longitude)  && ct.respond_to?(:description) &&
      identifier == ct.identifier  && latitude == ct.latitude &&
      longitude == ct.longitude   && description == ct.description         
    end
