    def read_packet
      begin
        lines = []
        while(str = @socket.readline.to_s) do
          break if str.strip == '.'
          lines << str.strip
        end
        parse_error(lines)
        lines
      rescue Errno::ETIMEDOUT, Errno::ECONNREFUSED, Errno::ECONNRESET, EOFError => ex
        raise Munin::ConnectionError, ex.message
      end
    end
