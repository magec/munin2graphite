    def open
      begin
        @socket = TCPSocket.new(@host, @port)
        @socket.sync = true
        welcome = @socket.gets
        unless welcome =~ /^# munin node at/
          raise Munin::AccessDenied
        end
        @connected = true
      rescue Errno::ETIMEDOUT, Errno::ECONNREFUSED, Errno::ECONNRESET => ex
        raise Munin::ConnectionError, ex.message
      rescue EOFError
        raise Munin::AccessDenied
      rescue Exception => ex
        raise Munin::ConnectionError, ex.message
      end
    end
